DROP TABLE IF EXISTS S
DROP TABLE IF EXISTS P
DROP TABLE IF EXISTS J
DROP TABLE IF EXISTS SPJ

CREATE TABLE S
(
SNO CHAR(9) PRIMARY KEY,
SNAME CHAR(20) UNIQUE,
STATUS INT,
CITY CHAR(20)
);

CREATE TABLE P
(
PNO CHAR(9) PRIMARY KEY,
PNAME CHAR(20),
COLOR CHAR(8),
WEIGHT INT
);

CREATE TABLE J
(
JNO CHAR(9) PRIMARY KEY,
JNAME CHAR(20),
CITY CHAR(20)
);

CREATE TABLE SPJ
(
SNO CHAR(9),
PNO CHAR(9),
JNO CHAR(9),
QTY INT
);
